  if (!sheetElement) {

  }
  if (sheetElement) {
    rtl = sheetElement.getAttribute('dir') == 'rtl';

    while (r && r != sheetElement) {
      t += r.offsetTop;
      r = r.offsetParent;
    }

      offsetX += c.offsetLeft;
      c = c.offsetParent;
    }

    }

    s.top = t + 'px';
    s.display = 'block';
    s.border = '1px solid #000000';
  }
